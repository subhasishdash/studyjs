How to read a PDF file in JavaScript?

Posted on 04th April 2016 at 9:00 PM


Reading PDF file using JavaScript is super easy and will be done using the following steps:

  • 1.Create a Canvas to render the PDF.
  • 2.Download required libraries.
  • 3.Read the PDF file.
  • 4.Rendering of PDF file to canvas.

1.Create a Canvas to render the PDF.

As the contents of the PDF will be rendered to the canvas ,so let's create the canvas first which is nothing but a basic HTML tag as shown below. 
<canvas id="canvas" width=512 height=512></canvas>
Or alternatively, if you want to append it dynamically, then use the below snippet 
var canvas = document.createElement('canvas');
canvas.width = 512;
canvas.height = 512;
document.body.appendChild(canvas);

2.Download required libraries.

After creating the canvas ,we need to download the required libraries which is required for parsing and reading a PDF file.
So lets navigate to https://github.com/mozilla/pdf.js to download the libraries. After downloading unzip it and look for src folder and add the folder to your project like the below screenshot.

3.Read the PDF file.

Next, lets read the pdf file . Now for this demo ,I will take a button where we can upload and the read in JavaScript.
So, lets create the input button first . 
<input type="file" id="files"/>
Now, lets add a event handler for the above button to listen for file read events. 
//add a handler for reading files from computer
var newFile = document.getElementById('files');
newFile.onchange = function (evt) {
	parseAndRenderPDF(evt, this);
}

4.Rendering of PDF file to canvas.

So, now whenever we upload a new file parseAndRenderPDF(evt); will be called which will parse and render the pdf to canvas. 
var parseAndRenderPDF = function (evt, that) {
	var canvas = document.getElementById('canvas');
	var tgt = evt.target || window.event.srcElement;
	var files = tgt.files;
	// FileReader support
	if (FileReader && files && files.length) {
		var fr = new FileReader();
		var extension = files[0].name.split('.').pop().toLowerCase();
		//if extension is not pdf ,then stop processing further and exit.
		if(extension !== 'pdf') {
			alert('please choose a PDF file');
			return;
		} else {
			//parse and process the PDF file.
			fr.onload = function(e) {
				console.debug("Loaded the PDF file");
				PDFJS.getDocument(e.target.result).then(function(pdf) {
						// Using promise to fetch the page
						pdf.getPage(1).then(function(page) {
							var scale = 1.5;
							var viewport = page.getViewport(scale);
							var context = canvas.getContext('2d');
							context.clearRect(0,0, canvas.width, canvas.height);
							canvas.height = viewport.height;
							canvas.width = viewport.width;
							var renderContext = {
									canvasContext: context,
									viewport: viewport
							};
							//Step 1: store a refer to the renderer
							var pageRendering = page.render(renderContext);
							//Step : hook into the pdf render complete event
							var completeCallback = pageRendering._internalRenderTask.callback;
							pageRendering._internalRenderTask.callback = function (error) {
								//Step 2: what you want to do before calling the complete method                  
								completeCallback.call(that, error);
							};
						});
					});	
			}
			fr.readAsDataURL(that.files[0]);
		}
	}
}
